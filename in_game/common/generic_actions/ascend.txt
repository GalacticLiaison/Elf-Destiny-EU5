# scope:actor = country, scope:recipient = religion
ascend = {
	type = religious

	sound = UI_action_religion_generic

	show_message = no #optional, to prevent messages showing
	show_message_to_target = no #optional, to prevent messages to show to the target
	
	ai_tick = never
	ai_tick_frequency = 24
	automation_tick = never
	automation_tick_frequency = 12
	player_automated_category = religiousdoctrines

	potential = {
		scope:recipient = {
			has_religious_influence = yes
		}
		scope:actor = {
			ruler ?= {
				is_any_elf_type = yes
			}
		}
	}

	allow = {
		trigger_if = {
			limit = {
				scope:actor = {
					ruler ?= {
						has_racial_trait = { trait = elf_tier_3 }
					}
				}
			}
			custom_tooltip = {
				text = grand_portal_has_been_restored
				has_global_variable = portal_restored
			}
		}
		trigger_if = {
			limit = {
				scope:actor = {
					ruler ?= {
						has_racial_trait = { trait = elf_tier_9 }
					}
				}
			}
			custom_tooltip = {
				text = cant_yet_enter_god_tier
				always = no
			}
		}
	}
	
	# The price will be paid in the follow up event
	should_execute_price = no
	price = price:ascension_base_price
	price_modifier = {
		value = 1
		if = {
			limit = {
				scope:actor = {
					ruler ?= {
						has_racial_trait = { trait = elf_tier_1 }
					}
				}
			}
			multiply = 0.5
		}
		else = {
			multiply = scope:actor.ruler.ascension_price_multiplier_by_elf_level
		}
	}

	select_trigger = {
		looking_for_a = religion
		source = actor
		target_flag = recipient
		name = "choose_religion"
		column = {
			data = name
		}
	}

	effect = {
		scope:actor = {
			trigger_event_silently = {
				id = ascension_events.1
			}
		}
	}

	ai_will_do = {
		add = -100
	}
}
