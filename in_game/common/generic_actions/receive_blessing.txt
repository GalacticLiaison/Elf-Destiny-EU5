# scope:actor = country, scope:recipient = religion
receive_blessing = {
	type = religious

	sound = UI_action_religion_generic

	show_message = no #optional, to prevent messages showing
	show_message_to_target = no #optional, to prevent messages to show to the target
	
	ai_tick = never
	ai_tick_frequency = 24
	automation_tick = never
	automation_tick_frequency = 12
	player_automated_category = religiousdoctrines

	
	potential = {
		scope:recipient = {
			has_religious_influence = yes
		}
		scope:actor = {
			ruler ?= {
				is_any_elf_type = yes
			}
		}
	}

	allow = {
		scope:actor = {

			trigger_if = {
				limit = {
					has_all_standard_blessings = yes
					NOT ={
						has_religious_aspect = religious_aspect:familial_familiarity
					}
				}

				custom_tooltip = {
					text = "secret condition"
					has_policy = doctrine_consanguinity_unrestricted
				}
			}
			trigger_else = {
				has_all_standard_blessings = no
			}
		}
	}
	
	# price = 0
	price = price:blessing_base_price
	price_modifier = {
		value = 1
		add = scope:actor.number_of_owned_blessings
	}

	select_trigger = {
		looking_for_a = religion
		source = actor
		target_flag = recipient
		name = "choose_religion"
		column = {
			data = name
		}
	}

	effect = {
		scope:actor = {
			trigger_event_non_silently = {
				id = ascension_events.2
			}
		}
	}

	ai_will_do = {
		add = -100
	}
}
