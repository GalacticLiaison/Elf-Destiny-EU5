pa_cosponser_expedition = {
	estate = calaquendi_estate
	
	potential = {
		NOT ={
			has_variable = expedition_underway
		}
	}
	
	on_accept = {
		add_gold = {
			value = expedition_cost
			multiply = 0.5
		}
		trigger_event_non_silently = {
			id = elf_destiny_expedition.1
		}
	}

	chance = 20
}

pa_push_towards_quality = {
	estate = calaquendi_estate
	
	potential = {
		societal_value:quality_vs_quantity > 0
	}
	
	on_accept = {
		add_government_power = government_power_mild_penalty
		change_societal_value = {
			type = quality_vs_quantity
			value = societal_value_minor_move_to_left
		}
	}
	chance = 10
}

pa_move_towards_quality = {
	estate = calaquendi_estate
	
	potential = {
		societal_value:quality_vs_quantity < 0
		societal_value:quality_vs_quantity > -90
	}
	
	on_accept = {
		add_government_power = government_power_mild_penalty
		change_societal_value = {
			type = quality_vs_quantity
			value = societal_value_tiny_move_to_left
		}
	}
	chance = 10
}
