has_any_elven_bloodline = {
	OR = {
		has_genetic_trait = { trait = royal_elf_valerith }
		has_genetic_trait = { trait = royal_elf_serelion }
		has_genetic_trait = { trait = royal_elf_gwynthorn }
		has_genetic_trait = { trait = royal_elf_thundarael }
		has_genetic_trait = { trait = royal_elf_daelurin }
		has_genetic_trait = { trait = royal_elf_lormelis }
	}
}

has_genetic_trait = {
	is_target_in_variable_list = {
		name = genetic_traits
		target = trait:$trait$
	}
}

has_racial_trait = {
	has_variable = character_race 
	var:character_race = trait:$trait$
}
