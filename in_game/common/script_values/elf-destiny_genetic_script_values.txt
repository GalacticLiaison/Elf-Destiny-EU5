# chance to start with a genetic trait

	value = 0
    add = {
		if = { 
			limit = {
				has_game_rule = genetic_traits_normal_rarity
			}
			value = 20
		}
	}
    add = {
		if = { 
			limit = {
				has_game_rule = genetic_traits_more_common
			}

		}
	}
    add = {
		if = { 
			limit = {
				has_game_rule = genetic_traits_more_rare
			}
			value = 10
		}
	}
}
