daughters_sent_to_become_aeluran_sisters_effect = {
	every_character = {
		limit = {
			is_female = yes
			character_age >= 3
			character_age < 7
			NOT = {
				has_variable = refused_aeluran_training
			}
		}

		save_scope_as = new_sister

		set_variable = {
			name = country_of_origin
			value = owner
		}


		if = {
			limit = {
				owner = { is_ai = no }
				is_child_of = owner.ruler
			}

			owner = {
				trigger_event_non_silently = {
					id = aeluran_events.4 
				}
			}
		}
		else = {
			set_child_education = child_education:aeluran_education
			move_country = c:DWD
		}

	}
}

daughters_return_as_aeluran_sisters_effect = {
	every_character = {
		limit = {
			is_female = yes
			is_adult = yes
			has_variable = country_of_origin
			NOT = {
				has_trait = aeluran_sister
			}
		}
		save_scope_as = new_sister

		become_aeluran_sister = yes

		move_country = var:country_of_origin
		remove_variable = country_of_origin

		if = {
			limit = {
				owner = { is_ai = no}
			}

			owner = {
				trigger_event_silently = {
					id = aeluran_events.41
				}
			}
		}
	}
}

find_aeluran_advisor_effect = {
	if = {
		limit = {
			any_character = {
				eligible_to_be_aeluran_advisor = yes
			}
		}
		random_character = {
			limit = {
				eligible_to_be_aeluran_advisor = yes
			}
			save_scope_as = aeluran_advisor
			set_variable = aeluran_advisor
		}

		save_scope_value_as = {
			name = aeluran_from_court
			value = yes
		}
	}
	else = {
		random_list = {
			1 = {
				modifier = {
					has_country_modifier = calaquendi_estate_blocked_from_cabinet
					add = -2
				}
				create_new_aeluran_advisor = { estate = calaquendi_estate }
			}
			2 = {
				modifier = {
					has_country_modifier = nobles_estate_blocked_from_cabinet
					add = -2
				}
				create_new_aeluran_advisor = { estate = nobles_estate }
			}
			2 = {
				modifier = {
					has_country_modifier = clergy_estate_blocked_from_cabinet
					add = -2
				}
				create_new_aeluran_advisor = { estate = clergy_estate }
			}
		}
	}
}

assign_new_aeluran_advisor = {
	if = {
		limit = {
			NOT = {
				exists = scope:aeluran_advisor
			}
		}
		
		find_aeluran_advisor_effect = yes
	}


	if = {
		limit = {
			NOT = {
				exists = scope:aeluran_from_court
			}
		}
		custom_tooltip = {
			text = aeluran_advisor_joins_realm

			scope:aeluran_advisor = {
				move_country = root
			}
		}
	}

	custom_tooltip = {
		text = aeluran_advisor_joins_and_become_head_of_cabinet
		aeluran_advisor_joins_cabinet = { aeluran_advisor_scope = scope:aeluran_advisor }
	}
}

aeluran_advisor_joins_cabinet = {
	if = {
		limit = {
			exists = $aeluran_advisor_scope$
		}

		$aeluran_advisor_scope$.owner = {
			if = {
				limit = { # Is the cabinet full?
					any_cabinet_character = {
						count >= this.modifier:government_size
					}
				}
				random_cabinet_character = {
					limit = {
						NOT = { this = $aeluran_advisor_scope$ }
					}
					save_temporary_scope_as = old_cabinet_member
				}
				remove_from_cabinet = scope:old_cabinet_member
			}

			add_character_to_random_open_cabinet = {
				target = $aeluran_advisor_scope$
			}

			promote_to_head_of_cabinet_effect = {
				target = $aeluran_advisor_scope$
			}
		}
	}
}

# root = character, scope:target = new country, scope:owner = old country
retain_aeluran_advisors_effect = {
	debug_log = "on_character_moved_country: retain_aeluran_advisors"
	if = {
		limit = {
			exists = scope:owner
		}

		set_variable = { name = aeluran_advisor_returned value = yes days = 15 }

		move_country = scope:owner

		if = {
			limit = {
				in_cabinet = no
			}
			aeluran_advisor_joins_cabinet = { aeluran_advisor_scope = root }
		}
	}
}

aeluran_succession_effect = {
	debug_log = "aeluran_succession_effect"
	if = {
		limit = {
			scope:target = {
				has_trait = aeluran_matron
			}
		}
		debug_log = "aeluran_succession_effect aeluran_matron"
		remove_list_global_variable = { name = aeluran_matrons target = scope:target }
		promote_random_sister_to_matron_effect = yes
	}
	if = {
		limit = {
			scope:target = {
				has_trait = aeluran_matriarch
			}
		}
		debug_log = "aeluran_succession_effect aeluran_matriarch"
		promote_random_matron_to_matriarch_effect = yes
	}
}

promote_random_sister_to_matron_effect = {
	debug_log = "promote_random_sister_to_matron_effect"

	random_country = {
		limit = {
			any_character = {
				has_trait = aeluran_sister
			}
		}
		random_character = {
			limit = {
				has_trait = aeluran_sister
			}
			save_scope_as = new_matron
		}
	}

	if = {
		limit = {
			NOT = {
				exists = scope:new_matron
			}
		}
		debug_log = "promote_random_sister_to_matron_effect fallback"

		create_new_aeluran_matron_fallback = yes
	}
	else = {
		scope:new_matron = {
			remove_aeluran_rank = {aeluran_rank = aeluran_sister}
			become_aeluran_matron = yes
		}
	}

	debug_log = "promote_random_sister_to_matron_effect matron selected"
	debug_log_scopes = yes

	if = {
		limit = {
			scope:new_matron.owner = {
				is_ai = no
			}
		}

		scope:new_matron.owner = {
			trigger_event_non_silently = {
				id = aeluran_events.2 
			}
		}
	}

	add_to_global_variable_list = {
		name = aeluran_matrons
		target = scope:new_matron
	}
}

promote_random_matron_to_matriarch_effect = {
	random_in_global_list = {
		variable = aeluran_matrons
		save_scope_as = new_matriarch
	}

	scope:new_matriarch = {
		remove_aeluran_rank = {aeluran_rank = aeluran_matron}
		become_aeluran_matriarch = yes
	}

	c:DWD = { set_new_ruler_no_update = scope:new_matriarch }

	remove_list_global_variable = { name = aeluran_matrons target = scope:new_matriarch }

	promote_random_sister_to_matron_effect = yes

	every_country = {
		limit = {
			is_ai = no
			religion = religion:aeluran
			NOT = {
				scope:new_matriarch.owner = this
			}
		}

		trigger_event_non_silently = {
			id = aeluran_events.3
		}
	}

	if = {
		limit = {
			scope:new_matriarch = {
				owner = {
					is_ai = no
				}
			}
		}

		scope:new_matriarch = {
			owner = {
				trigger_event_non_silently = {
					id = aeluran_events.31
				}
			}
		}
	}
}

become_aeluran_sister = {
	add_trait = trait:aeluran_sister
	assign_aeluran_modifier = {aeluran_trait = aeluran_sister}
}

become_aeluran_matron = {
	add_trait = trait:aeluran_matron
	assign_aeluran_modifier = {aeluran_trait = aeluran_matron}
}

become_aeluran_matriarch = {
	add_trait = trait:aeluran_matriarch
	assign_aeluran_modifier = {aeluran_trait = aeluran_matriarch}
}

assign_aeluran_modifiers_effect = {
	assign_aeluran_modifier = {aeluran_trait = aeluran_sister}
	assign_aeluran_modifier = {aeluran_trait = aeluran_matron}
	assign_aeluran_modifier = {aeluran_trait = aeluran_matriarch}
}

assign_aeluran_modifier = {
	if = {
		limit = {
			has_trait = $aeluran_trait$
			NOT = { has_character_modifier = $aeluran_trait$} 
		}

		add_character_modifier = {
			modifier = $aeluran_trait$
			# mode = add_and_extend
		}
	}
}

remove_aeluran_rank = {
	if = {
		limit = {
			has_trait = $aeluran_rank$
		}
		remove_trait = trait:$aeluran_rank$
	}
	if = {
		limit = {
			has_character_modifier = $aeluran_rank$
		}
		remove_character_modifier = $aeluran_rank$
	}
}
