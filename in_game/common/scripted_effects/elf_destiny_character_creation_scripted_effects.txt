create_bloodline_orphan_effect = {
	random_list = {
		50 = {
			create_bloodline_orphan = {
				is_female = yes
				culture = $culture$
				blood = $blood$
			}
		}
		50 = {
			create_bloodline_orphan = {
				is_female = no
				culture = $culture$
				blood = $blood$
			}
		}
	}

	scope:orphan = {
		assign_genetic_trait = { trait = intellect_good_2 }
	}
}
