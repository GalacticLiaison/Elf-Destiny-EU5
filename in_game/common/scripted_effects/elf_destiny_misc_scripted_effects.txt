
setup_elf_culture_locations_effect = {
	culture_group:elven_culture_group = {
		every_country_in_culture_group = {
			save_temporary_scope_as = elf_country

			# These need to be seperate every_pop loops for some reason
			every_pop = {
				change_pop_religion = religion:aeluran
			}
			every_pop = {
				change_pop_culture = prev.culture
			}

			every_province = {
				every_location_in_province = {
					boost_starting_pops_of_elf_countries = yes
					add_some_pops_of_random_elf_cultures = yes
					increase_elf_food_supply = yes
				}
			}

			if = {
				limit = { NOT = { religion = religion:aeluran } }
				change_religion = religion:aeluran
			}
			if = {
				limit = { NOT = { has_estate_privilege = estate_privilege:aeluran_advisors } }
				grant_estate_privilege = estate_privilege:aeluran_advisors
			}
		}
	}
}

increase_elf_food_supply = {
	if = {
		limit = {
			OR = {
				raw_material = goods:wild_game
				raw_material = goods:fish
			}	
		}

		change_max_raw_material_workers = 3

		add_pop = {
			culture = owner.ruler.culture
			religion = owner.ruler.religion
			type = pop_type:laborers
			size = { value = 3}
		}

		if = {
			limit = {
				location_rank = location_rank:rural_settlement
				NOT = { has_building = building_type:forest_village }
			}
			construct_building = {
				building_type = building_type:forest_village
				cost_multiplier = 0.0
				cost_multiplier_reason = "game_start"
				instant = yes
			}
		}
	}

	if = {
		limit = {
			OR = {
				raw_material = goods:fur
			}	
		}

		change_max_raw_material_workers = 1

		add_pop = {
			culture = owner.ruler.culture
			religion = owner.ruler.religion
			type = pop_type:laborers
			size = { value = 1}
		}

		if = {
			limit = {
				location_rank = location_rank:rural_settlement
				NOT = { has_building = building_type:forest_village }
			}
			construct_building = {
				building_type = building_type:forest_village
				cost_multiplier = 0.0
				cost_multiplier_reason = "game_start"
				instant = yes
			}
		}
	}
}

start_elf_destiny_storyline_effect = {
	debug_log = "start_elf_destiny_storyline_effect"

	if = {
		limit = {
			any_country = {
				is_ai = no
			}
		}

		debug_log = "Player countries exist!"
	}
	else = {
		debug_log = "Player countries DONT exist!"
	}

	random_country = {
		limit = {
			# is_ai = no
		}
		debug_log = "start_elf_destiny_storyline_effect country reached"

		trigger_event_non_silently = {
			id = elf_destiny_story.99
			days = 2
		}	
	}
}

register_aeluran_leadership = {
	if = {
		limit = {
			has_trait = aeluran_matron
		}
		add_to_global_variable_list = { name = aeluran_matrons target = this }
	}
}

seed_calaquendi_estate_characters_effect = {
	if = {
		limit = {
			culture = { has_culture_group = culture_group:elven_culture_group}
		}
		create_calaquendi_estate_character = {is_female = yes scope_name = new_calaquendi}
		create_calaquendi_estate_character = {is_female = no scope_name = new_calaquendi}
	}
}

set_initial_consanguinity_law = {
	if = {
		limit = {
			NOT = {
				has_law = consanguinity_law
			}
		}
		add_policy = policy:doctrine_consanguinity_cousins
	}
}

populate_calaquendi_estate_with_characters_effect = {
	if = {
		limit = {
			characters_in_calaquendi_estate_count < 3
		}

		if = {
			limit = {
				any_character = {
					is_adult = yes
					has_estate = estate_type:nobles_estate
					has_racial_trait = { trait = elf_tier_2 }
				}
				NOT = {
					has_variable = noble_ascended
				}
			}
			trigger_event_silently = {
				id = elf_destiny_misc.42
			}
		}
		else = {
			trigger_event_silently = {
				id = elf_destiny_misc.41 
			}
		}

	}
}

set_calaquendi_education_effect = {
	if = {
		limit = {
			is_adult = no
			has_child_education = balanced_education
		}

		set_child_education = child_education:calaquendi_apprenticeship_education
	}
}

assign_enchantress_trait_effect = {
	every_character = {
		limit = {
			is_adult = yes
			num_of_traits_of_category = { type = ruler value < 3 }
			NOR = { 
				has_trait = enchantress
				has_trait = enchanter
			}
		}
		random = {
			chance = 10

			save_scope_as = recipient

			if = {
				limit = {
					is_ruler = yes
					owner = {
						is_ai = no
					}
				}
				owner = {
					trigger_event_non_silently = {
						id = elf_destiny_misc.20
					}
				}
			}
			else = {
				if = {
					limit = {
						is_female = yes
					}
					add_trait = trait:enchantress
				}
				else = {
					add_trait = trait:enchanter
				}
			}
		}
	}
}

assign_heroic_trait_effect = {
	every_character = {
		limit = {
			is_adult = yes
			num_of_traits_of_category = { type = ruler value < 3 }
			NOT = { has_trait = heroic }
		}
		random = {
			chance = 10

			save_scope_as = recipient

			if = {
				limit = {
					is_ruler = yes
					owner = {
						is_ai = no
					}
				}
				owner = {
					trigger_event_non_silently = {
						id = elf_destiny_misc.21
					}
				}
			}
			else = {
				add_trait = trait:heroic
			}
		}
	}
}

record_dead_rulers_dynasty_effect = {
	debug_log = "record_dead_rulers_dynasty"
	if = {
		limit = {
			exists = scope:old_ruler.dynasty
		}
		set_variable = {
			name = previous_ruling_dynasty
			value = scope:old_ruler.dynasty
		}
	}
