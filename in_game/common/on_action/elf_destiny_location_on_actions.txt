on_location_changed_owner = {
    on_actions = {
		manage_building_modifiers
	}
}

manage_building_modifiers = {
    trigger = {
        scope:loser = {
            OR = {
                is_elven_country = yes
                religion = religion:aeluran
            }
        }
    }

    effect = {
        debug_log = "manage_building_modifiers"
        debug_log_scopes = yes
    }
}
