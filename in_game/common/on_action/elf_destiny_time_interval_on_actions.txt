assign_aeluran_advisor = {
	trigger = {
		ruler ?= {
			is_any_elf_type = yes
		}
		religion = religion:aeluran
		has_estate_privilege = estate_privilege:aeluran_advisors
		NOT = {
			any_cabinet_character = {
				has_variable = aeluran_advisor
			}
		}
	}
	effect = {				
		trigger_event_non_silently = {
			id = aeluran_events.1
		}			
	}
}
