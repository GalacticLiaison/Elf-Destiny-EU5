scripted_effect assign_race_to_character = {
    scope:recipient = {
        remove_any_racial_trait = yes
        assign_racial_trait = { trait = elf_tier_$elf_level$ }
    }
    trigger_event_silently = { id = elf_destiny_debug.2 }
}

scripted_effect swap_tiered_genetic_trait = {
    scope:recipient = {
        if = {
            limit = {
                has_genetic_trait = { trait = $selected_trait$ }
            }
            remove_genetic_trait = { trait = $selected_trait$  }
        }
        else = {
            remove_all_genetic_traits_of_type = { trait_type = $type$ }
            assign_genetic_trait = { trait = $selected_trait$ }
        }
    }
}

scripted_effect toggle_genetic_trait = {
    scope:recipient = {
        if = {
            limit = {
                has_genetic_trait = { trait = $selected_trait$ }
            }
            remove_genetic_trait = { trait = $selected_trait$  }
        }
        else = {
            assign_genetic_trait = { trait = $selected_trait$ }
        }
    }
}
