elf_destiny_misc.20 = {
	type = country_event
	title = elf_destiny_misc.20.title
	desc = elf_destiny_misc.20.desc
	trigger = {

	}

	image = "gfx/interface/illustrations/event/backgrounds/divine_spark.dds"

	immediate = {
		if = {
			limit = {
				scope:recipient = {
					is_female = yes
				}
			}
			trait:enchantress = {
				save_scope_as = trait
			}
		}
		else = {
			trait:enchanter = {
				save_scope_as = trait
			}
		}
	}
	
	option = {
		name = elf_destiny_misc.20.a

		if = {
			limit = {
				scope:recipient = {
					is_female = yes
				}
			}
			scope:recipient = {
				add_trait = trait:enchantress
			}
		}
		else = {
			scope:recipient = {
				add_trait = trait:enchanter
			}
		}
	}
}
