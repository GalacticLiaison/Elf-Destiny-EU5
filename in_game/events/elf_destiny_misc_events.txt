				create_bloodline_family_effect = { culture = astrion_culture blood = thundarael}
			}
			10 = {
				create_bloodline_family_effect = { culture = atlareth_culture blood = lormelis}
			}
		}
	}
	
	option = {
		name = elf_destiny_misc.31.accept

		scope:father_lord = {
			move_country = root
		}

			move_country = root
		}
		scope:older_daughter = {
			move_country = root
		}
		scope:younger_son = {

		}
	}

	option = {
		name = elf_destiny_misc.31.refuse

		custom_tooltip = {
			text = elf_destiny_misc.31.family_leaves
			kill_character_silently = {
				target = scope:father_lord
				reason = vanished
			}
			kill_character_silently = {
				target = scope:mother
				reason = vanished
			}
			kill_character_silently = {
				target = scope:older_daughter
				reason = vanished
			}

# Calaquendi Estate Population Events
elf_destiny_misc.41 = { # Two new members join
	type = country_event
	title = elf_destiny_misc.41.title
	desc = elf_destiny_misc.41.desc
	trigger = {

	}

	image = "gfx/interface/illustrations/event/backgrounds/green_leaves.dds"

	immediate = {
		create_calaquendi_estate_character = {is_female = yes scope_name = new_calaquendi_1}
		create_calaquendi_estate_character = {is_female = no scope_name = new_calaquendi_2}
	}
	
	option = {
		name = elf_destiny_misc.41.a

		scope:new_calaquendi_1 ?= {
			move_country = root
		}

		scope:new_calaquendi_2 ?= {
			move_country = root
		}
	}
}

elf_destiny_misc.42 = { # Noble Ascends to High Elf and is raised to Calaquendi
	type = country_event
	title = elf_destiny_misc.42.title
	desc = elf_destiny_misc.42.desc
	trigger = {
		any_character = {
			is_adult = yes
			has_estate = estate_type:nobles_estate
			has_racial_trait = { trait = elf_tier_2 }
		}
	}

	image = "gfx/interface/illustrations/event/backgrounds/green_leaves.dds"

	immediate = {
		random_character = {
			limit = {
				is_adult = yes
				has_estate = estate_type:nobles_estate
				has_racial_trait = { trait = elf_tier_2 }
			}
			save_scope_as = new_calaquendi
		}

		scope:new_calaquendi = {
			ascension_effect = yes
		}

		set_variable = { name = noble_ascended value = yes years = 15 }
	}
	
	option = {
		name = elf_destiny_misc.42.a
		scope:new_calaquendi ?= {
			change_character_estate = estate_type:calaquendi_estate
		}
	}
}
