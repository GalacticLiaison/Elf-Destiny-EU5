namespace = ascension_events

ascension_events.1 = {
	type = country_event
	title = ascension_events.1.title
	desc = {
		first_valid = {
			triggered_desc = {
				desc = ascension_events.1.desc.family
				trigger = {
					exists = scope:portrait_2
				}
			}
			desc = ascension_events.1.desc.solo
		}
		triggered_desc = {
			desc = ascension_events.1.desc.cant_afford_family
			trigger = {
				religious_influence < ascension_price_family_positive
			}
		}
	}

	image = "gfx/interface/illustrations/event/backgrounds/green_leaves.dds"

	# illustration_tags = {
    #     10 = regular
    #     10 = interior
    # }

	immediate = {
		ruler = {
			save_scope_as = portrait_1
		}
		if = {
			limit = {
				ruler = {
					num_of_spouses > 0
				}
			}
			ruler = {
				first_spouse = {
					if = {
						limit = {
							is_alive = yes
						}
					}
					save_scope_as = portrait_2
				}
			}
		}
		else_if = {
			limit = {
				ruler = {
					num_of_children > 0
				}
			}
			ruler = {
				ordered_child = {   #get the youngest one
					limit = {
						is_alive = yes
						age_in_years > 0
					}
					order_by = {
						value = age_in_years
					}
					max = 1
					save_scope_as = portrait_2
				}
			}
			save_scope_value_as = {
				name = family_ascension_price
				value = ascension_price_family_positive
			}
		}
	}

	option = {
		name = ascension_events.1.default
		add_religious_influence = ascension_price_solo

		trigger = {
			NOT = {
				exists = scope:portrait_2
			}
		}

		save_scope_value_as = {
			name = ascension_type
			value = flag:solo
		}
		custom_tooltip = {
			text = you_ascend_alone
		}
		
		trigger_event_silently = {
			id = ascension_events.11
		}

	}
	
	option = {
		name = ascension_events.1.solo
		add_religious_influence = ascension_price_solo

		trigger = {
			exists = scope:portrait_2
		}

		scope:portrait_1 = {
			save_scope_as = portrait_2
		}

		save_scope_value_as = {
			name = ascension_type
			value = flag:solo
		}
		custom_tooltip = {
			text = you_ascend_alone
		}

		trigger_event_silently = {
			id = ascension_events.11
		}

	}

	option = {
		name = ascension_events.1.family
		add_religious_influence = ascension_price_family

		trigger = {
			exists = scope:portrait_2
			religious_influence >= ascension_price_family_positive
		}

		custom_tooltip = {
			text = you_ascend_with_all_immediate_family_members
			save_scope_value_as = {
				name = ascension_type
				value = flag:family
			}
		}

		trigger_event_silently = {
			id = ascension_events.11
		}

	}

	option = {
		name = ascension_events.1.changed_my_mind
	}
}

scripted_effect choose_blessing_option = {
	random_list = {
		0 = {
			modifier = {
				add = 100
				NOT = {
					has_religious_aspect = religious_aspect:the_rut
				}
				NOT = {
					AND = {
						exists = scope:blessing_option_1
						scope:blessing_option_1 = flag:the_rut
					}
				}
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:the_rut
			}
		}
		0 = {
			modifier = {
				add = 100
				NOT = {
					has_religious_aspect = religious_aspect:noble_husbandry
				}
				NOT = {
					AND = {
						exists = scope:blessing_option_1
						scope:blessing_option_1 = flag:noble_husbandry
					}
				}
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:noble_husbandry
			}
		}
		0 = {
			modifier = {
				add = 100
				NOT = {
					has_religious_aspect = religious_aspect:heroic_courage
				}
				NOT = {
					AND = {
						exists = scope:blessing_option_1
						scope:blessing_option_1 = flag:heroic_courage
					}
				}
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:heroic_courage
			}
		}
		0 = {
			modifier = {
				add = 100
				has_policy = doctrine_consanguinity_unrestricted
				NOT = {
					has_religious_aspect = religious_aspect:familial_familiarity
				}
				NOT = {
					AND = {
						exists = scope:blessing_option_1
						scope:blessing_option_1 = flag:familial_familiarity
					}
				}
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:familial_familiarity
			}
		}
		0 = {
			modifier = {
				add = 100
				NOT = {
					has_religious_aspect = religious_aspect:beguiling_nature
				}
				NOT = {
					AND = {
						exists = scope:blessing_option_1
						scope:blessing_option_1 = flag:beguiling_nature
					}
				}
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:beguiling_nature
			}
		}
		1 = { # fallback
			modifier = {
				add = -1
				has_religious_aspect = religious_aspect:noble_husbandry
				has_religious_aspect = religious_aspect:heroic_courage
				has_religious_aspect = religious_aspect:beguiling_nature
				has_religious_aspect = religious_aspect:the_rut
			}
			save_scope_value_as = {
				name = blessing_option_$option$
				value = flag:none
			}
		}
	}
}

scripted_effect receive_blessing = {
	if = {
		limit = {
			scope:blessing_option_$option$ = flag:the_rut
		}
		add_religious_aspect = religious_aspect:the_rut
	}
	else_if = {
		limit = {
			scope:blessing_option_$option$ = flag:noble_husbandry
		}
		add_religious_aspect = religious_aspect:noble_husbandry
	}
	else_if = {
		limit = {
			scope:blessing_option_$option$ = flag:heroic_courage
		}
		add_religious_aspect = religious_aspect:heroic_courage
	}
	else_if = {
		limit = {
			scope:blessing_option_$option$ = flag:beguiling_nature
		}
		add_religious_aspect = religious_aspect:beguiling_nature
	}
	else_if = {
		limit = {
			scope:blessing_option_$option$ = flag:familial_familiarity
		}
		add_religious_aspect = religious_aspect:familial_familiarity
	}
}
