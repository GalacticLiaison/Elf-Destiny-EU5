	image = "gfx/interface/illustrations/event/backgrounds/aeluran_leaves.dds"

	trigger = {

	}

	immediate = {
		# if = {
		# 	limit = {
		# 		scope:new_matron ?= {
		# 			is_ruler = yes
		# 		}
		# 	}
		# }
	}

	option = {
		name = aeluran_events.4.refuse

		trigger = {
			NOT = {
				has_estate_privilege = estate_privilege:daughters_of_the_red
			}
		}

		scope:new_sister = {
			set_variable = refused_aeluran_training
		}

		add_religious_influence = religious_influence_severe_penalty
	}

	option = {
		name = aeluran_events.4.a

		trigger = {

		}

		scope:new_sister = {
			set_child_education = child_education:aeluran_education
			move_country = c:DWD
		}

		add_religious_influence = religious_influence_severe_bonus
	}
}
